$$\tag{1}
w = mg.
$$

$$\tag{2}
\mu_k = 0 \rightarrow f_k = \mu_k N = 0.
$$

$$\tag{3}
v_x(0) = v_{0x}.
$$

$$\tag{4}
v_y(0) = 0.
$$

$$\tag{5}
\begin{array}{rcl}
\sum F_y & = & 0 \newline
N - w & = & 0 \newline
N & = & w.
\end{array}
$$

$$\tag{6}
\begin{array}{rcl}
\sum F_x & = & ma_x \newline
-f_k & = & ma_x \newline
0 & = & ma_x \newline
a_x & = & 0.
\end{array}
$$

$$\tag{7}
x = x_0 + v_{0x} t.
$$

$$\tag{8}
y = y_0.
$$
